    Set Controle_HVAC_Dorm1 = 1,  !- A7
    Set Controle_VN_Dorm1 = 0,  !- A8

    Set Controle_HVAC_Dorm1 = 0,  !- A10

    Set Controle_VN_Dorm1 = 0,  !- A14

    Set Controle_VN_Dorm1 = 0,  !- A16
    ENDIF,                   !- A17
